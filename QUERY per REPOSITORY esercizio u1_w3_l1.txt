Es_1 = SELECT * FROM clienti WHERE nome = 'mario'

Es_2 = SELECT nome, cognome FROM clienti WHERE anno_di_nascita = 1982

Es_3 = SELECT COUNT(*) FROM fatture WHERE importo = 20;

Es_4 = SELECT data_attivazione FROM prodotti
WHERE EXTRACT(YEAR FROM data_attivazione) = 2017
AND (in_produzione = true OR in_commercio = true);

Es_5 = SELECT * FROM fatture
JOIN clienti ON fatture.id_cliente = numero_cliente
WHERE importo < 1000 

Es_6 = SELECT fatture.*, fornitori.denominazione FROM fatture
JOIN fornitori USING (numero_fornitore)

Es_7 = SELECT EXTRACT(YEAR FROM data_fattura) AS anno, COUNT(*) AS numero_fatture FROM fatture
WHERE iva = 20
GROUP BY EXTRACT(YEAR FROM data_fattura)

Es_8 = SELECT EXTRACT(YEAR FROM data_fattura), COUNT(*) AS numero_fatture , SUM(importo) AS importo_totale FROM fatture
GROUP BY (EXTRACT(YEAR FROM data_fattura))

Es_9 = SELECT EXTRACT(YEAR FROM data_fattura)
FROM fatture
WHERE tipologia ILIKE 'td01%'

Es_10 = SELECT clienti.regione_residenza, SUM(importo) AS importo_totale FROM fatture
JOIN clienti on numero_cliente = fatture.id_cliente
GROUP BY regione_residenza

Es_11 = SELECT COUNT(*) from clienti 
JOIN fatture ON id_cliente = clienti.numero_cliente
WHERE importo > 50 AND clienti.anno_di_nascita = 1980
